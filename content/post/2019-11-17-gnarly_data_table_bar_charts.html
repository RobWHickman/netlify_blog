city_data <- city_data %>%
  #select out relevant columns
  select(country_id = Country.Code, country = Country.or.area,
         city_id = City.Code, city = Urban.Agglomeration,
         X1950, X1955, X1960, X1965, X1970, X1975, X1980, X1985, X1990,
         X1995, X2000, X2005, X2010, X2015, X2020, X2025, X2030) %>%
  #melt the data to long format
  reshape2::melt(id.vars = c("country_id", "country", "city_id", "city"),
                 variable.name = "year", value.name = "population") %>%

  #extract the english names for cities
  mutate(city_name = case_when(
    grepl("\\(", city) ~ str_extract(city,  "(?<=\\().+?(?=\\))"),
    grepl("-", city) ~ gsub("-.*", "", city),
    TRUE ~ as.character(city)
  ))

#start all teams with an ELO of 1000
team_ratings <- data.frame(team = unique(match_data$home),
                           rating = 1000)

#function to calculate team ELO and update back to the ratings df
calc_ELO <- function(row) {
  #get the difference in ratings
  hr <- team_ratings$rating[which(team_ratings$team == row$home)]
  vr <- team_ratings$rating[which(team_ratings$team == row$visitor)]
  dr <- (hr + 100) - vr
  
  e_result <- 1/ (10^(-dr/400)+1)
  new_hr <- hr + ((row$K*row$G) * (row$result - e_result))
  new_vr <- vr + ((row$K*row$G) * ((1-row$result) - (1-e_result)))
  
  team_ratings$rating[which(team_ratings$team == row$home)] <<- new_hr
  team_ratings$rating[which(team_ratings$team == row$visitor)] <<- new_vr
  
  output_row <- row %>%
    mutate(h_rating = new_hr, v_rating = new_vr)
  
  return(output_row)
}

  #start by merging in all the "missing" days
  #ie. days where matches were not played

  #complete by team in dataset

  #delete the completion of NA data

  #set as DT for munging
  setDT()

munged_football_dt <- munged_football_data %>%
  #order by date
  .[order(date)] %>%
  #find the first game in any league for each season
  .[.[!is.na(season), .I[which.min(date)], by = c("season")]$V1, actual_season := season] %>%
  #fill down
  .[order(date, actual_season), actual_season := na.locf(actual_season, na.rm = FALSE)]
#subset season breaks out
munged_football_dt <- munged_football_dt %>%
  #label each game a team players in a season
  .[!is.na(rating), game := 1:.N, by = c("team", "actual_season")] %>%
  #find the first game each team plays in a season
  .[.[, .I[which.min(game)], by = c("team", "actual_season")]$V1, first_game := 1] %>%
  #find the earliest game which is labelled as the first game in a season
  #this is the 'start' of that season
  .[.[first_game == 1, .I[which.min(date)], by = c("actual_season")]$V1, first_date := 1] %>%
  #we'll use a binary 'keep' variable when we delete days in between seasons
  #wil lfill this down later
  .[date %in% unique(date[which(!is.na(.$first_date))]), keep := 1] %>%
  #do the same with the last game played in any division in a season
  .[.[, .I[which.max(game) + 7], by = c("team", "actual_season")]$V1, last_game := 1] %>%
  .[.[last_game == 1, .I[which.max(date)], by = c("actual_season")]$V1, last_date := 1] %>%
  .[date %in% unique(date[which(!is.na(.$last_date))]), keep := 0] %>%
  #fill down the keep value and keep only those with a 1
  .[, keep := na.locf(keep, na.rm = TRUE), by = team] %>%
  .[keep == 1]
#tween the ratings between matches
munged_football_dt <- munged_football_dt %>%
  #make sure in date order
  .[order(date)] %>%
  #use tween fill for each team to get intermediate ratings
  .[, rating := tween_fill(rating, ease = "linear"), by = team] %>%
  #for each teams last season in the data frame will be some days overhanging
  #fill these down as rating obviously won't change
  .[actual_season == max(actual_season), rating := na.locf(rating, na.rm = FALSE), by = team] %>%
  #label each unique date as another frame grouped by season
  .[, frame := .GRP, by = date] %>%
  .[frame %% 5 == 1] %>%
  #take the rolling mean over 10 days to smooth out jumps
  .[order(date), roll_rating := frollmean(rating, n = 10, algo = "exact", align = "left"), by = "team"] %>%
  #remove unrated rows
  .[!is.na(roll_rating)]
#use decimal labelling for days in each season
munged_football_dt <- munged_football_dt %>%
  .[frame != lag(frame, default = 0), season_frame := 1:.N, by = actual_season] %>%
  #fill down
  .[, season_frame := na.locf(season_frame)] %>%
  #frame is the season plus the amount through that season
  #could probably do something fancy and label as unique days but this suffices
  .[, season_frame := actual_season + season_frame/(max(season_frame) + 1)]
